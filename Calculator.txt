public int multiply(int a, int b){
    int result = 0;

    if (a < 0 && b < 0){
        for (int i = a; i <= -1; i++)
            result-=b;    
    }
    else if (a < 0){
        for (int i = 1; i <= b; i++)
            result+=a;
    }
    else if (b < 0){
        for (int i = 1; i <= a; i++)
            result+=b;
    }
    else {
        for (int i = 1; i <= b; i++)
            result+=a;    
    }
    return result;
}
